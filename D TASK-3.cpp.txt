#include <iostream>
using namespace std;

int main() {
    int arr[4] = {1, 2, 3, 4};

    for (int i = 0; i < 2; i++) {
        int temp = arr[i];
        arr[i] = arr[3 - i];
        arr[3 - i] = temp;
    }

    cout << "Reversed array: ";
    for (int i = 0; i < 4; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
